daaa zzzz xxcc ckck

  d = Direction   0 = Lesen, 1 = Schreiben
aaa = Destination 0..6 (7 = reserved)

zzzz = Anzahl Daten 0..8 mit adress 0..4
  xx = reserve = 0

 cc = 0: Daten
      1: Adresse + Daten
      2: Info
      3: Flash Interface
